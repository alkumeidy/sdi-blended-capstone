users {
	id int pk
	task_id int > tasks.id
	admin boolean
	username varchar unique
	password varchar
	logged_in boolean
}

tasks {
	id int pk
	title varchar
	description varchar
	due_date date
	completed boolean
	significant boolean
	date_created timestamp
	date_modified timestamp
}

msl {
	id int pk
	title varchar
	description varchar
	user_id int > users.id
	date_created timestamp
}

tags {
	id int pk
	task_id int > tasks.id
	name varchar
	msl_id int > msl.id
}
